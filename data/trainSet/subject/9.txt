function setImageSize() {
	var nMaxWidth = 700;
	var arEls = document.getElementsByTagName("IMG");
	for( var i=0;i<arEls.length;i++ ){
		if( arEls[i].width >nMaxWidth ) {	//如果图片宽度大于nMaxWidth，则等比缩小显示
			var nHeight = (arEls[i].height * (nMaxWidth/arEls[i].width));
			arEls[i].height = nHeight;
			arEls[i].width  =	nMaxWidth;
			arEls[i].style.height = nHeight;
			arEls[i].style.width = nMaxWidth;
		}
	}
}

function GetActiveText(objHTML) {

		var obj=document.getElementById(objHTML);
		obj.currPos = document.selection.createRange().duplicate();
	}

function InsertText(objHTML,strText,bolReplace) {
	if (strText=="") {
		return "";
	}
	var obj=document.getElementById(objHTML);
	if (!obj) {
		return "";
	}
	if(document.selection){
		if (obj.currPos){
			if(bolReplace && (obj.value=="")){
				obj.currPos.text=strText;
			}
			else{
				obj.currPos.text+=strText;
			}
		}
		else{
			obj.value+=strText;
		}
	}
	else{
		if(bolReplace){
			obj.value=obj.value.slice(0,obj.selectionStart) + strText + obj.value.slice(obj.selectionEnd,obj.value.length);
		}
		else{
			obj.value=obj.value.slice(0,obj.selectionStart) + strText + obj.value.slice(obj.selectionStart,obj.value.length);
		}
	}
}
function ReplaceText(objHTML,strPrevious,strNext) {
	var obj=document.getElementById(objHTML);
	if (!obj) {
		return "";
	}
	var strText;
	if(document.selection && document.selection.type == "Text"){
		if (obj.currPos){
			var range = document.selection.createRange();
			range.text = strPrevious + range.text + strNext;
			return "";
		}
		else{
			strText=strPrevious + strNext;
			return strText;
		}
	}
	else{
		if(obj.selectionStart || obj.selectionEnd){
			strText=strPrevious + obj.value.slice(obj.selectionStart,obj.selectionEnd) + strNext;
			return strText;
		}
		else{
			strText=strPrevious + strNext;
			return strText;
		}
	}
}

2010年12月03日 08:12:41 　来源： 新华国际 <br>
<br>
<br>
<br>
法新社华盛顿12月1日电 俄罗斯总理弗拉基米尔・普京今天说，俄罗斯没有将战术核武器转移到北约盟国附近，并指责西方在此问题上导致紧张局势升级。<br>
<br>
在美国有线电视新闻国际公司(CNN)的《拉里・金访谈》节目中，普京说：“我们没有把导弹前移到你们领土附近。”<br>
<br>
美国官员称，俄罗斯几个月前将战术核武器转移到了距北约国家的边境仅几英里的地方。<br>
<br>
普京说，西方国家正“计划在我们的边境、我们的领土附近部署导弹”，以防止所谓的伊朗核活动的威胁。把导弹部署在俄罗斯边境附近的可能性“当然让我们感到担心”，如果出现这种情况，“我们不得不采取一些针对性的行动”。<br>
<br>
俄罗斯总统梅德韦杰夫最近警告说，如果俄罗斯和西方不能就欧洲新的导弹防御计划达成一致，可能引发“新一轮军备竞赛”，莫斯科将部署新的武器系统。<br>
<br>
普京则说，一旦将来俄罗斯的安全面临新威胁，莫斯科“将不得不通过不同的途径确保自己的安全”。<br>
<br>
   俄新社莫斯科12月1日电 普京在接受CNN采访时表示，如果美国不批准俄美两国总统2010年4月签署的。《新削减和限制进攻性战略武器条约》的话，那么俄罗斯将被迫开始增加核武器。<br>
 <br>

<br>
普京在接受CNN采访时说，美国国防部长罗伯特・盖茨对俄罗斯民主的评价是严重错误的。他还说，一些美国总统得到的全民选票少于对手，只是因为赢得了选举入团的多数票才得以上台执政。<br>
<br>
普京说：“当我们与美国朋友交谈并告诉他们美国存在制度问题时，他们会说：‘别干预我们的事务。’我也想对美国同事提出建议：不要干预俄罗斯人民的最高选择。”<br>
<br>
俄新社莫斯科12月1日电 普京在接受CNN采访时称，维基揭秘网公布的对他与梅德韦杰夫总统关系的评价既粗鲁又不道德。<br>
<br>
维基揭秘网公布了一批美国外交人员的公函，其中提到了俄罗斯总统梅德韦杰夫和总理普京，把他们比作漫画角色罗宾与蝙蝠侠。<br>
<br>
普京说：“问题的实质在于，这是我们的相互协作问题，它是国家内政的重要因素。但是，坦白地说，我们从没有料到，这（对相互协作的评价）会是如此傲慢、粗鲁和不道德的。”<br>
