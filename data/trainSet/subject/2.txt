function countStringLength(str){
	var str_length = 0;
	for (var i=0;i<str.length;i++){
		if (str.charCodeAt(i)<=256){
			str_length +=1;
		}else{
			str_length +=2;
		}
	}
	return str_length;
}
function cutStringToLength(str,length){
	var tmp_str = str;
	for (var i=0;i<str.length ;i++ ) {
		if (countStringLength(tmp_str) > length){
			tmp_str = tmp_str.substring(0,tmp_str.length-1);
		}else{
			break;
		}
	}
	return tmp_str;
}
